program factorial{
int n;

function int factRecursion(int n){

    if (n == 1){
        return 1;
    }

    else{
        return n * factRecursion(n - 1);
    }
}


function int factNoRecursion (int n){

    int f, i;

    f = 1;
    i = 1;

    while(i<= n){
        f = f * i;
        i ++;
    }
    return f;

}


main(){
    n = 5;

  int result;

  result = factNoRecursion(n);

  print(result);
}


}
